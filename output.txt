\item{\textbf{Show instructor with 2nd max sal in instructor relation. }
\\
\\
\textbf{Command - }
\begin{lstlisting}[language=SQL]
SELECT * 
FROM instructor 
WHERE SAL = (
    SELECT MAX(SAL) 
    FROM instructor 
    WHERE SAL < ( 
        SELECT MAX(SAL) FROM instructor 
        ) 
    );
\end{lstlisting}

\item{\textbf{Increase budget by 10% which is having budget less then avg budget.}
\\
\\
\textbf{Command - }
\begin{lstlisting}[language=SQL]
UPDATE department
SET BUDGET = 1.1*BUDGET
WHERE BUDGET < (SELECT AVG(BUDGET) FROM department);
\end{lstlisting}
